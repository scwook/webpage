Raspberry Pi DHT22 Device Support Development

본 문서는 DHT22 Sensor에 대한 EPICS Library개발을 목표로 한다.
테스트에 사용된 DHT22 Sensor는DFRobot의 Sensor 모듈로 DHT22 Sensor에 필요한 저항을 보드에 부착하여 Connector로 연결 가능하게 만든 제품이다. 만약 Sensor만 사용할 경우 5k ohme의 pull-up resistor를 VDD와 Data사이에 추가로 연결해야 한다. 참고로 DHT22는 DHT11보다 정밀도가 향상된 Sensor이다.

테스트를 위한 하드웨어 구성은 다음과 같다.

  • Raspberry Pi Model B+
  • DFRobot DHT22 Sensor

최종 목표는 다음 같은 dht22.db를 만들어 온도 및 습도 값을 읽는 것이다.

record(dht, "DHT22:DHT")
{
  field(DTYP, "DHT22")
  field(INP, "@1")
  field(SCAN, "2 second")
  field(FLNK, "DHT22:FAN")
}

record(fanout, "DHT22:FAN")
{
  field(LNK1, "DHT22:TEM")
  field(LNK2, "DHT22:HUM")
}

record(calc, "DHT22:TEM")
{
  field(INPA, "DHT22:DHT.TEM")
  field(CALC, "A")
  field(PREC, "2")
}

record(calc, "DHT22:HUM")
{
  field(INPA, "DHT22:DHT.HUM")
  field(CALC, "A")
  field(PREC, "2")
}

위 DB에서 새롭게 정의된 dht record type은 온도 값을 나타내는 TEM field와 습도 값을 나타내는 HUM field를 가지고 있다. 이러한 2개의 field는 fanout을 통해 각각 DHT22:TEM, DHT22:HUM PV로 분리되어 읽어 진다.

DHT22 Sensor는 하나의 Line으로 신호를 전송하는 1-wire 통신을 한다. 1-wire 통신은 기본적으로 timing에 맞춰 데이터를 전송하는데 이러한 timing 정보는 1-wire protocol에 정의되어 있으나 정확한 timing 정보를 알기 위해서는 제품 Manual을 참고해야 한다. 참고로 1-wire은 Raspberry Pi의 GPIO Pin과 연결한다.

typedef struct dhtset { /* dht input dset */
	long		number;
	DEVSUPFUN	dev_report;

static long init_record(void *precord,int pass)
{
    dhtRecord	*prec = (dhtRecord *)precord;
    dhtdset	*pdset;
    long	status;

    if (pass==0) return(0);

    if(!(pdset = (dhtdset *)(prec->dset))) {
	recGblRecordError(S_dev_noDSET,(void *)prec,"dht: init_record");
	return(S_dev_noDSET);
    }
    /* must have read_dht function defined */
    if( (pdset->number < 5) || (pdset->read_dht == NULL) ) {
	recGblRecordError(S_dev_missingSup,(void *)prec,"dht: init_record");
	return(S_dev_missingSup);
    }

    if( pdset->init_record ) {
	if((status=(*pdset->init_record)(prec))) return(status);
    }
    return(0);
}

static long process(void *precord)
{
	dhtRecord	*prec = (dhtRecord *)precord;
	dhtdset		*pdset = (dhtdset *)(prec->dset);
	long		 status;
	unsigned char    pact=prec->pact;

	if( (pdset==NULL) || (pdset->read_dht==NULL) ) {
		prec->pact=TRUE;
		recGblRecordError(S_dev_missingSup,(void *)prec,"read_dht");
		return(S_dev_missingSup);
	}

	/* pact must not be set until after calling device support */
	status=(*pdset->read_dht)(prec);
	/* check if device support set pact */
	if ( !pact && prec->pact ) return(0);
	prec->pact = TRUE;

	recGblGetTimeStamp(prec);
	/* check for alarms */
	checkAlarms(prec);
	/* check event list */
	monitor(prec);
	/* process the forward scan link record */
        recGblFwdLink(prec);

	prec->pact=FALSE;
	return(status);
}

static long get_alarm_double(DBADDR *paddr,struct dbr_alDouble *pad)
{
    dhtRecord	*prec=(dhtRecord *)paddr->precord;
    int		fieldIndex = dbGetFieldIndex(paddr);

    if(fieldIndex == dhtRecordVAL) {
        pad->upper_alarm_limit = prec->hhsv ? prec->hihi : epicsNAN;
        pad->upper_warning_limit = prec->hsv ? prec->high : epicsNAN;
        pad->lower_warning_limit = prec->lsv ? prec->low : epicsNAN;
        pad->lower_alarm_limit = prec->llsv ? prec->lolo : epicsNAN;
    } else recGblGetAlarmDouble(paddr,pad);
    return(0);
}

static void checkAlarms(dhtRecord *prec)
{
	double		val;
	float		hyst, lalm, hihi, high, low, lolo;
	unsigned short	hhsv, llsv, hsv, lsv;

	if(prec->udf == TRUE ){
		recGblSetSevr(prec,UDF_ALARM,INVALID_ALARM);
		return;
	}
	hihi = prec->hihi; lolo = prec->lolo; high = prec->high; low = prec->low;
	hhsv = prec->hhsv; llsv = prec->llsv; hsv = prec->hsv; lsv = prec->lsv;
	val = prec->val; hyst = prec->hyst; lalm = prec->lalm;

	/* alarm condition hihi */
	if (hhsv && (val >= hihi || ((lalm==hihi) && (val >= hihi-hyst)))){
	        if (recGblSetSevr(prec,HIHI_ALARM,prec->hhsv)) prec->lalm = hihi;
		return;
	}

	/* alarm condition lolo */
	if (llsv && (val <= lolo || ((lalm==lolo) && (val <= lolo+hyst)))){
	        if (recGblSetSevr(prec,LOLO_ALARM,prec->llsv)) prec->lalm = lolo;
		return;
	}

	/* alarm condition high */
	if (hsv && (val >= high || ((lalm==high) && (val >= high-hyst)))){
	        if (recGblSetSevr(prec,HIGH_ALARM,prec->hsv)) prec->lalm = high;
		return;
	}

	/* alarm condition low */
	if (lsv && (val <= low || ((lalm==low) && (val <= low+hyst)))){
	        if (recGblSetSevr(prec,LOW_ALARM,prec->lsv)) prec->lalm = low;
		return;
	}

	/* we get here only if val is out of alarm by at least hyst */
	prec->lalm = val;
	return;
}

static void monitor(dhtRecord *prec)
{
	unsigned short	monitor_mask;
	double		delta;

        monitor_mask = recGblResetAlarms(prec);
	/* check for value change */
	delta = prec->mlst - prec->val;
	if(delta<0.0) delta = -delta;
	if (delta > prec->mdel) {
		/* post events for value change */
		monitor_mask |= DBE_VALUE;
		/* update last value monitored */
		prec->mlst = prec->val;
	}

	/* check for archive change */
	delta = prec->alst - prec->val;
	if(delta<0.0) delta = -delta;
	if (delta > prec->adel) {
		/* post events on value field for archive change */
		monitor_mask |= DBE_LOG;
		/* update last archive value monitored */
		prec->alst = prec->val;
	}

	/* send out monitors connected to the value field */
	if (monitor_mask){
		db_post_events(prec,&prec->val,monitor_mask);
	}
	return;
}

dht Record가 만들어 졌으면 Record Process에 의해 호출 되는 Device Support 코드를 만들어야 한다. 결과적으로 Device Support 코드는 Sensor로 부터 값을 읽고 Record에 전달하는 코드로 구성이 되어 있다. 코드 작성을 위해 devDHT22.c 파일을 만들고 다음과 같이 작성한다. 

#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#include <epicsExport.h>

#include <recSup.h>
#include <recGbl.h>

#define MAX_TIME 85

static long init_record(dhtRecord *pdht);
static long read_dht(dhtRecord *pdht);

typedef struct Dht_Info
{
  CALLBACK *callback;

  int pin;

  double temp;
  double humi;

  double pre_temp;
  double pre_humi;

}DHT_INFO;

void readDHT(DHT_INFO *dht_info);

int dht_val[5];

static void myCallback(CALLBACK *pcallback)
{
  dhtRecord *precord;
  struct rset *prset;

  callbackGetUser(precord, pcallback);
  prset = (struct rset *)(precord->rset);

  dbScanLock((dbCommon*)precord);
  (*prset->process)(precord);
  dbScanUnlock((dbCommon*)precord);
}

static long init_record(dhtRecord *pdht)
{
  DHT_INFO *dht_info = malloc(sizeof(DHT_INFO));
  CALLBACK *pcallback = (CALLBACK *)(calloc(1, sizeof(CALLBACK)));

  callbackSetCallback(myCallback, pcallback);
  callbackSetPriority(priorityLow, pcallback);
  callbackSetUser(pdht, pcallback);

  if(wiringPiSetup() == -1)
    return 1;

  int i;
  for(i=0;i<5;i++)
    dht_val[i] = 0;

  int pin = atoi(pdht->inp.value.instio.string);

  dht_info->pin = pin;
  dht_info->callback = pcallback;

  pdht->dpvt = dht_info;

  return 0;
}

static long read_dht(dhtRecord *pdht)
{
  DHT_INFO *dht_info = pdht->dpvt;
  CALLBACK *pcallback = (CALLBACK *)dht_info->callback;

  if(pdht->pact)
  {
    readDHT(dht_info);

    pdht->val = dht_info->temp;
    pdht->tem = dht_info->temp;
    pdht->hum = dht_info->humi;

    pdht->udf = FALSE;

    return 2;
  }
  else
  {
    pdht->pact = TRUE;
    callbackRequestDelayed(pcallback, 0.5);

    return 0;
  }

}

struct
{
  long num;
  DEVSUPFUN     report;
  DEVSUPFUN     init;
  DEVSUPFUN     init_record;
  DEVSUPFUN     get_ioint_info;
  DEVSUPFUN     read_dht;
  DEVSUPFUN     special_linconv;
} devDhtAsyn = {
  6,
  NULL,
  NULL,
  init_record,
  NULL,
  read_dht,
  NULL
};

epicsExportAddress(dset,devDhtAsyn);

void readDHT(DHT_INFO *dht_info)
{
  int pin = dht_info->pin;

  int i=0;
  for(i=0;i<5;i++)
      dht_val[i]=0;

  epicsUInt8 lststate=HIGH;
  epicsUInt8 counter=0;

  pinMode(pin,OUTPUT);
  digitalWrite(pin,LOW);
  delay(18);
  digitalWrite(pin,HIGH);
  delayMicroseconds(40);
  pinMode(pin,INPUT);

  int j=0;
  for(i=0;i=4)&&(i%2==0))
    {
      dht_val[j/8]<<=1;
      if(counter>16)
        dht_val[j/8]|=1;
      j++;
    }
  }
  float t = 0.0f;
  float h = 0.0f;
  if((j>=40)&&(dht_val[4]==((dht_val[0]+dht_val[1]+dht_val[2]+dht_val[3])& 0xFF)))
  {
    h = (float)dht_val[0] * 256 + (float)dht_val[1];
    h /= 10;

    dht_info->humi = h;
    dht_info->pre_humi = h;

    t = (float)(dht_val[2] & 0x7F)* 256 + (float)dht_val[3];
    t /= 10.0;

    if ((dht_val[2] & 0x80) != 0) t *= -1;

    dht_info->temp = t;
    dht_info->pre_temp = t;
  }
  else
  {
      dht_info->humi = dht_info->pre_humi;
      dht_info->temp = dht_info->pre_temp;
  }
}

devDHT22.c 코드는 크게 초기활르 위한 init_record와 Sensor로 부터 값을 읽는 read_dht 함수로 이루어져 있다.

init_record 에서는 INP field를 통해 전달받은 GPIO Pin 번호를 정수형으로 변환한다. 여기서 atoi한수는 문자열을 정수형으로 변환하는 C Library 함수이다.

read_dht에서는 Sensor로 부터 값을 읽어 Record에 저장하는데 실제 Sensor에서 값을 읽는 과정은 readDHT함수에서 진행된다.

다음은 앞서 작성한 record와 device 코드를 연결시켜 주기위한 devDHT22.dbd 파일을 만든다.

device(dht, INST_IO, devDhtAsyn, "DHT22")

device는 순서대로 Record Type, Link Type, Device Support Struct, Device Type을 인자로 각각의 의미는 다음과 같다.

        Record Type: 사용하는 Record 이름, 여기>서는 새로만든 dht를 사용한다.
         Link Type: EPICS Base에 정의된 Link Type중 하나, INST_IO는 문자열 하나를 받는 Link Type 이다.
        Device Support Struct: 함수 주소를 가진 구>조체 이름, devDHT22.c에서 사용한 구조체 이름을 적는다.
        Device Type: DTYP field에서 사용할 이름, 보 통 Protocol 또는 Device 이름을 사용한다.
